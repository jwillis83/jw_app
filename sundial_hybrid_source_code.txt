import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { IonicModule } from '@ionic/angular';

interface GroupedHistory {
  monthName: string;
  logs: any[];
}

@Component({
  selector: 'app-history',
  templateUrl: './history.page.html',
  styleUrls: ['./history.page.scss'],
  standalone: true,
  imports: [CommonModule, IonicModule]
})
export class HistoryPage implements OnInit {
  public groupedSections: GroupedHistory[] = [];

  ngOnInit() {
    this.loadAndGroupHistory();
  }

  ionViewWillEnter() {
    this.loadAndGroupHistory();
  }

  loadAndGroupHistory() {
    const data = localStorage.getItem('sundial_logs');
    if (!data) {
      this.groupedSections = [];
      return;
    }

    try {
      const rawLogs = JSON.parse(data);
      const groups: { [key: string]: any[] } = {};

  
      rawLogs.forEach((log: any) => {
        if (log && log.date) {

          const dateObj = new Date(log.date.replace(' (Tracked)', ''));
          const monthYearLabel = dateObj.toLocaleString('default', { month: 'long', year: 'numeric' });

          if (!groups[monthYearLabel]) {
            groups[monthYearLabel] = [];
          }
          groups[monthYearLabel].push(log);
        }
      });

     
      this.groupedSections = Object.keys(groups).map(key => ({
        monthName: key,
        // Reverse individual items so newest records display first
        logs: groups[key].reverse()
      })).reverse();

    } catch (e) {
      console.error('Error parsing history tracking arrays', e);
      this.groupedSections = [];
    }
  }

  deleteLog(sectionIndex: number, logIndex: number) {
    const targetItem = this.groupedSections[sectionIndex].logs[logIndex];
    const data = localStorage.getItem('sundial_logs');
    
    if (data) {
      let logs = JSON.parse(data);
      // Find exact item match inside original master array and remove it
      const matchIndex = logs.findIndex((l: any) => l.date === targetItem.date);
      if (matchIndex !== -1) {
        logs.splice(matchIndex, 1);
        localStorage.setItem('sundial_logs', JSON.stringify(logs));
        this.loadAndGroupHistory(); // Refresh view
      }
    }
  }
}

import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { IonicModule } from '@ionic/angular';
import { FormsModule } from '@angular/forms';
import { RouterModule } from '@angular/router';

@Component({
  selector: 'app-settings',
  templateUrl: './settings.page.html',
  styleUrls: ['./settings.page.scss'],
  standalone: true,
  imports: [CommonModule, IonicModule, FormsModule, RouterModule]
})
export class SettingsPage implements OnInit {
  // Configuration States & Default Values
  public gpsAlwaysOn: boolean = false;
  public gpsOnlyWhenUsing: boolean = true;
  public localStorageActive: boolean = true;
  public cloudStorageActive: boolean = false;
  public allowKeepHistory: boolean = true;
  public historyDaysLimit: number = 30;

  ngOnInit() {
    this.loadSettings();
  }

  // Handle Radio Toggle exclusivity for GPS settings
  toggleGpsSetting(type: string) {
    if (type === 'always' && this.gpsAlwaysOn) {
      this.gpsOnlyWhenUsing = false;
    } else if (type === 'using' && this.gpsOnlyWhenUsing) {
      this.gpsAlwaysOn = false;
    }
    this.saveSettings();
  }

  // Save Configs to LocalStorage
  saveSettings() {
    const configData = {
      gpsAlwaysOn: this.gpsAlwaysOn,
      gpsOnlyWhenUsing: this.gpsOnlyWhenUsing,
      localStorageActive: this.localStorageActive,
      cloudStorageActive: this.cloudStorageActive,
      allowKeepHistory: this.allowKeepHistory,
      historyDaysLimit: this.historyDaysLimit
    };
    localStorage.setItem('sundial_settings', JSON.stringify(configData));
  }

  loadSettings() {
    const saved = localStorage.getItem('sundial_settings');
    if (saved) {
      const data = JSON.parse(saved);
      this.gpsAlwaysOn = data.gpsAlwaysOn;
      this.gpsOnlyWhenUsing = data.gpsOnlyWhenUsing;
      this.localStorageActive = data.localStorageActive;
      this.cloudStorageActive = data.cloudStorageActive;
      this.allowKeepHistory = data.allowKeepHistory;
      this.historyDaysLimit = data.historyDaysLimit;
    }
  }

  // Purge Tracking History Button Action
  purgeHistory() {
    localStorage.removeItem('sundial_logs');
    alert('Tracking history logs have been completely purged.');
  }
}

import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { IonicModule } from '@ionic/angular';
import { Router } from '@angular/router';

@Component({
  selector: 'app-storage',
  templateUrl: './storage.page.html',
  styleUrls: ['./storage.page.scss'],
  standalone: true,
  imports: [CommonModule, IonicModule]
})
export class StoragePage implements OnInit {
  public selectedDate: string = '';
  public minDate: string = '';
  public maxDate: string = '';

  constructor(private router: Router) {}

  ngOnInit() {
    this.calculate30DayWindow();
  }


  calculate30DayWindow() {
    const today = new Date();
    const thirtyDaysAgo = new Date();
    thirtyDaysAgo.setDate(today.getDate() - 30);

    
    this.maxDate = today.toISOString().split('T')[0];
    this.minDate = thirtyDaysAgo.toISOString().split('T')[0];
    this.selectedDate = this.maxDate; // Default selection to today
  }


  onDateSelected(event: any) {
    const dateValue = event.detail.value;
    if (dateValue) {

      const parsedDate = new Date(dateValue).toLocaleDateString();      

      localStorage.setItem('sundial_active_target_date', parsedDate);

      this.router.navigate(['/track']);
    }
  }
}

import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { IonicModule } from '@ionic/angular';
import { Geolocation } from '@capacitor/geolocation';
import { Motion } from '@capacitor/motion';

@Component({
  selector: 'app-track',
  templateUrl: './track.page.html',
  styleUrls: ['./track.page.scss'],
  standalone: true,
  imports: [CommonModule, IonicModule]
})
export class TrackPage implements OnInit {
  // Live Phone States
  public pitch: number = 0;
  public roll: number = 0;
  public heading: number = 0;

  // Calculated Target Sun Position
  public sunElevation: number = 0; // Altitude angle above horizon
  public sunAzimuth: number = 0;   // Compass heading of the sun

  // Calculated Targets for Phone
  public targetPitch: number = 0;
  public targetRoll: number = 0;

  // Visual Offsets for the Bubble Circle
  public bubbleX: number = 0;
  public bubbleY: number = 0;

  public isAligned: boolean = false;
  public logHistory: any[] = [];
  private trackingInterval: any;

  ngOnInit() {
    this.startGPS();
    this.startSensors();
    this.loadLogs();
  }

  // Calculate Sun Position via Solar Geometry
  calculateSunPosition(lat: number, lon: number) {
    const now = new Date();
    
    // 1. Day of the year (1-365)
    const start = new Date(now.getFullYear(), 0, 0);
    const diff = now.getTime() - start.getTime();
    const oneDay = 1000 * 60 * 60 * 24;
    const dayOfYear = Math.floor(diff / oneDay);

    // 2. Solar Declination (Angle between sun rays and earth equator)
    const declination = 23.45 * Math.sin((2 * Math.PI / 365) * (dayOfYear - 80));

    // 3. True Solar Time / Hour Angle
    const hours = now.getHours() + now.getMinutes() / 60 + now.getSeconds() / 3600;
    // Standard LSTM approximation (assuming approximate 15 degrees per timezone)
    const localTimeMeridian = Math.round(lon / 15) * 15;
    const timeCorrection = 4 * (lon - localTimeMeridian); 
    const solarTime = hours + (timeCorrection / 60);
    const hourAngle = 15 * (solarTime - 12);

    // Convert degrees to radians for JS Math processing
    const latRad = lat * (Math.PI / 180);
    const decRad = declination * (Math.PI / 180);
    const hrRad = hourAngle * (Math.PI / 180);

    // 4. Calculate Sun Elevation (Altitude above horizon)
    const sinElevation = Math.sin(latRad) * Math.sin(decRad) + Math.cos(latRad) * Math.cos(decRad) * Math.cos(hrRad);
    this.sunElevation = Math.asin(sinElevation) * (180 / Math.PI);

    // 5. Calculate Sun Azimuth (Compass direction of sun)
    const cosAzimuth = (Math.sin(decRad) - Math.sin(latRad) * sinElevation) / (Math.cos(latRad) * Math.cos(Math.asin(sinElevation)));
    let azAngle = Math.acos(cosAzimuth) * (180 / Math.PI);
    
    // Adjust compass heading based on morning vs afternoon
    this.sunAzimuth = (hourAngle > 0) ? 360 - azAngle : azAngle;

    // 6. Convert Sun Path to Target Phone Tilts
    // When phone lays flat on panel:
    // To face sun perpendicular, phone needs to tilt back by (90 - SunElevation)
    const idealTiltAngle = 90 - this.sunElevation;

    // Map that tilt angle against the compass difference between sun direction and phone direction
    const headingDifference = (this.heading - this.sunAzimuth) * (Math.PI / 180);
    
    this.targetPitch = idealTiltAngle * Math.cos(headingDifference);
    this.targetRoll = idealTiltAngle * Math.sin(headingDifference);

    // Round calculations for clean interface displays
    this.sunElevation = Math.round(this.sunElevation);
    this.sunAzimuth = Math.round(this.sunAzimuth);

    this.processBubbleLevel();
  }

  async startGPS() {
    try {
      const position = await Geolocation.getCurrentPosition();
      const lat = position.coords.latitude;
      const lon = position.coords.longitude;

      // Recalculate sun position every 4 seconds to match the moving sun path
      this.calculateSunPosition(lat, lon);
      this.trackingInterval = setInterval(() => {
        this.calculateSunPosition(lat, lon);
      }, 4000);

    } catch (e) {
      console.error('GPS tracking failed', e);
    }
  }

  async startSensors() {
    try {
      await Motion.addListener('orientation', (event) => {
        this.pitch = event.beta ? Math.round(event.beta) : 0;
        this.roll = event.gamma ? Math.round(event.gamma) : 0;
      });

      window.addEventListener('deviceorientationabsolute', (event: any) => {
        if (event.alpha !== null) {
          this.heading = Math.round(360 - event.alpha);
        }
      }, true);
    } catch (e) {
      console.error('Sensors blocked', e);
    }
  }

  processBubbleLevel() {
    // Determine how far the user's real tilt is from the sun target angle
    this.bubbleY = (this.pitch - this.targetPitch);
    this.bubbleX = (this.roll - this.targetRoll);

    // App is perfectly perpendicular to the sun if difference is less than 3.5 degrees
    this.isAligned = Math.abs(this.bubbleY) < 3.5 && Math.abs(this.bubbleX) < 3.5;
  }

  saveLog() {
    const targetedDate = localStorage.getItem('sundial_active_target_date');
    const displayDate = targetedDate ? `${targetedDate} (Tracked)` : new Date().toLocaleString();

    const item = { 
      date: displayDate, 
      pitch: this.pitch, 
      roll: this.roll,
      azimuth: this.sunAzimuth,
      elevation: this.sunElevation
    };
    
    this.logHistory.push(item);
    if (this.logHistory.length > 30) this.logHistory.shift();
    localStorage.setItem('sundial_logs', JSON.stringify(this.logHistory));
    localStorage.removeItem('sundial_active_target_date');
    
    alert(`Perpendicular coordinates logged successfully!`);
  }

  loadLogs() {
    const data = localStorage.getItem('sundial_logs');
    if (data) this.logHistory = JSON.parse(data);
  }

  ngOnDestroy() {
    if (this.trackingInterval) clearInterval(this.trackingInterval);
  }
}
